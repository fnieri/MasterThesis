    \node at (-2.5, 0) [square, draw] (s) {\tiny $A$};
    \node at (0.5, 0) [square, draw] (eb) {\tiny $B$};

    \draw (cir.east) -- (eb.west);
    \draw (eb.east) -- (b.west);
    \draw (b.east) -- (ez.west);
